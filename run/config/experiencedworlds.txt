/*The maximum amount of time spent finding a viable spawn position. By
 default the vanilla world border is set to 0,0. Very often this can be in
 the middle of the ocean, the mod will try to find a valid spawn as close
 to 0,0 as it can.*/
fairnessCheckMaximumTime = 60
announcements = [
	"ADVANCEMENTS"
	"BORDER_GROWTH"
]
/*This config is balanced for vanilla minecraft. Modifying one value may
 put everything out of balance.*/
gameplay = {
/*	The base the logarithmic function should use. When set to 2, the border
	 will grow when stats reach a power of 2, so: 1,2,4,8,16,..., 10 is the
	 default. If you change this setting its effect cannot be reversed on
	 worlds where you play with it.*/
	logBase = 10
//	The max world border size
	maximumBorderSize = 2147483647
/*	The amount of time in milliseconds the border takes to grow 0.5 blocks
	 on every edge (aka increase width by 1 block)*/
	borderGrowthSpeed = 1000
//	The maximum advancement multiplier that can be reached
	advancementMultiplierMax = 50.0
/*	When one of these stats reaches a power of ten (including 1) the border
	 will grow. These are the only supported options: [minecraft:{mined,
	 crafted, used, broken, picked_up, dropped, killed, killed_by}]. There might
	 be other stats that work but I wouldn't recommend trying them as it is
	 untested.*/
	activeStats = [
		minecraft:mined
		minecraft:killed
	]
//	The multiplier base for advancements.
	advancementMultiplierBase = 1.01
/*	Rewards a border growth when a stat reaches 1 for the first time. Worlds
	 are permanently influenced by this option, changing it will not grant
	 or ungrant previous stat awards.*/
	awardOne = true
/*	The amount the world grows each time a stat point reaches the required
	 amount.*/
	sizeGained = 1.0
//	The starting world border size.
	startingSize = 1.0
/*	If true the multiplier gained per achievement will be equal to
	 advancementMultiplierBase^numAdvancements, if false:
	 advancementMultiplierBase*numAdvancements*/
	multipliersExponentialGain = true
/*	When enabled, the border will grow when a stat reaches a power of 10,
	 when disabled the border will grow everytime a stat grows by 1. If you
	 change this setting, its effects cannot be reversed on worlds where you
	 play with it.*/
	logarithmicStatRequirement = true
}
